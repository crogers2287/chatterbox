# Updated Nginx Proxy Manager Configuration for Production

## In NPM, update your proxy host with this Advanced configuration:

# Force cache refresh for all assets
location ~* \.(js|css|html)$ {
    proxy_pass http://192.168.1.195:5173;
    proxy_http_version 1.1;
    proxy_set_header Host $host;
    proxy_set_header Cache-Control "no-cache, no-store, must-revalidate";
    proxy_set_header Pragma "no-cache";
    
    # Add cache-busting headers
    add_header Last-Modified $date_gmt;
    add_header Cache-Control "no-cache, no-store, must-revalidate";
    add_header Pragma "no-cache";
    add_header Expires "0";
    
    # Force revalidation
    if_modified_since off;
    expires off;
    etag off;
}

# Inject API configuration into HTML
location = / {
    proxy_pass http://192.168.1.195:5173;
    proxy_http_version 1.1;
    proxy_set_header Host $host;
    
    # Inject script to force API URL
    sub_filter '</head>' '<script>window.VITE_API_URL="/api";console.log("API URL forced to /api");</script></head>';
    sub_filter_once on;
    sub_filter_types text/html;
}

# API proxy (updated)
location /api {
    proxy_pass http://192.168.1.195:6095;
    proxy_http_version 1.1;
    
    # Remove /api prefix
    rewrite ^/api/(.*) /$1 break;
    rewrite ^/api$ / break;
    
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
    
    # CORS headers
    add_header 'Access-Control-Allow-Origin' '*' always;
    add_header 'Access-Control-Allow-Methods' 'GET, POST, OPTIONS' always;
    add_header 'Access-Control-Allow-Headers' 'Content-Type' always;
    
    # Handle OPTIONS
    if ($request_method = 'OPTIONS') {
        add_header 'Access-Control-Allow-Origin' '*';
        add_header 'Access-Control-Allow-Methods' 'GET, POST, OPTIONS';
        add_header 'Access-Control-Allow-Headers' 'Content-Type';
        add_header 'Access-Control-Max-Age' 1728000;
        add_header 'Content-Type' 'text/plain; charset=utf-8';
        add_header 'Content-Length' 0;
        return 204;
    }
    
    proxy_connect_timeout 600s;
    proxy_send_timeout 600s;
    proxy_read_timeout 600s;
}
